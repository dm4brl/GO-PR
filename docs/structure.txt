/cmd
    /app-server          # Главный исполняемый файл сервера (API + cron + очереди)
/internal
    /api                 # HTTP и gRPC серверы, роутинг
    /domain              # Бизнес-логика, интерфейсы
        /user            # Домены пользователей
        /device          # Домены устройств
        /schedule        # Домены расписаний
        /automation      # Логика шаббата/йом тов автоматизации
    /repository          # Реализация доступа к данным
        /postgres        # База данных PostgreSQL
        /redis           # Кэш и очереди (Asynq)
    /service             # Интеграции и сторонние сервисы
        /mqtt            # Работа с MQTT брокером и клиентами
        /zmanim          # Клиент REST для получения шаббатных времен
    /worker              # Асинхронные задачи и очереди (Asynq workers)
/pkg                     # Переиспользуемые библиотеки и утилиты
    /logger              # Обертка над логированием
    /config              # Загрузка конфигурации (Viper / env)
/scripts                 # Скрипты деплоя, миграции и т.д.
/deploy                  # Docker, Kubernetes и CI/CD манифесты
/tests                   # Интеграционные тесты
/build                   # Билды, артефакты
/docs                    # Документация проекта
